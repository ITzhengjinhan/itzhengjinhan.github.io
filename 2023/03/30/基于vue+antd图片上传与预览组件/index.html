基于vue+antd图片上传，支持预览

vue
vue
发布日期:   2023-03-30

背景：现存的上传组件不能预览图片，放大缩小等操作

效果

上传组件

<template>
  <div class="upload-container" style="margin-top: 4px">
    <span v-for="(f, index) in fileList" :key="f.uid" class="img-content">
      <a-spin :spinning="f.loading">
        <img alt="" :src="f.url" @error="onImageError(index)" />
        <span class="delete">
          <span v-if="canEdit" class="common" @click.stop="deleteImg(f.uid)"> 删除 </span>
          <span v-if="canEdit" style="color: black; margin: 0 3px">|</span>
          <span class="common" @click.stop="previewImg(index)">预览</span>
        </span>
      </a-spin>
    </span>
    <a-upload
      v-if="canEdit && ((maxCount > 0 && fileList.length < maxCount) || !maxCount)"
      :multiple="multiple"
      :list-type="listType"
      :show-upload-list="false"
      :name="name"
      :data="extraInfo"
      :before-upload="beforeUpload"
    >
      <slot>
        <div v-if="(maxCount > 0 && fileList.length < maxCount) || !maxCount">
          <a-icon type="plus" />
          <div class="ant-upload-text">点击上传</div>
        </div>
      </slot>
    </a-upload>
    <picture-preview v-if="ppVisible" v-model="ppVisible" :imgs="fileList" :show-index="showIndex" />
  </div>
</template>

<script>
import remove from 'lodash/remove';
import PicturePreview from './PicturePreview';
/**
 * 图片上传
 *
 * 描述：支持图片上传和预览
 */
export default {
  name: 'UploadImgs',
  components: {
    PicturePreview,
  }
  model: {
    prop: 'val',
    event: 'valChange',
  },
  props: {
    // true: 编辑; false: 非编辑（场景：直接设置组件为编辑模式，此时operateType优先级较低）
    isEdit: {
      type: Boolean,
      default: false,
    },
    val: {
      type: Array,
      default: () => [],
    },
    /**
     * 附加信息
     */
    extraInfo: {
      type: Object,
      default: () => {},
    },
    // 多格式逗号隔开
    accept: {
      type: Array,
      default: () => ['jpg', 'jpeg', 'gif', 'png'],
    },
    listType: {
      type: String,
      default: 'picture-card',
      validator: (value) => {
        return ['text', 'picture', 'picture-card'].includes(value);
      },
    },
    name: {
      type: String,
      default: 'file',
    },
    multiple: {
      type: Boolean,
      default: false,
    },
    // 上传最大文件数
    maxCount: {
      type: Number,
    },
    // 文件最大限制 unit: MB
    maxSize: {
      type: Number,
      default: 5,
    },
  },
  data() {
    return {
      fileList: [],
      ppVisible: false,
      showIndex: 0,
    };
  },
  computed: {
    canEdit() {
      return this.isEdit;
    },
  },
  watch: {
    val: {
      handler(val) {
        this.fileList = (val || []).map((v) => ({ ...v, loading: false }));
      },
      immediate: true,
    },
  },
  methods: {
    // 自定义上传
    beforeUpload(file) {
      const fileInfo = {
        uid: file.uid,
        name: file.name,
        type: file.type,
        percent: 0,
        loading: true,
        size: file.size,
        url: '',
      };
      const { size, name, type } = file;
      const fileType = file.name.substring(file.name.lastIndexOf('.') + 1);
      if (this.accept.length > 0 && !this.accept.includes(fileType)) {
        this.$message.error(`图片只支持：${this.accept.join('，')}格式`);
        return false;
      }
      // 最大上传数量有值的情况
      if (!!this.maxCount) {
        if (this.fileList.length >= this.maxCount) {
          this.$message.warning(`最多上传${this.maxCount}张图片`);
          return false;
        }
      }
      if (size > this.maxSize * 1024 * 1024) {
        this.$message.warning(`${name}文件大小不能超过${this.maxSize}MB，请重新上传`);
        return false;
      }
      // 图片本地预览地址
      if (file.type.indexOf('image') > -1) {
        fileInfo.url = window.URL.createObjectURL(file);
      }
      this.fileList = [...this.fileList, fileInfo];
      this.uploadFile(file);
      return false;
    },
    // 普通上传
    uploadFile(file) {
      const fileIndex = this.fileList.findIndex((f) => f.uid === file.uid);
      if (fileIndex >= 0) {
        this.fileList[fileIndex].loading = false;
      }
      this.$emit('valChange', this.fileList);
      this.$emit('change', this.fileList);
      const formData = new FormData();
      formData.append(this.name, file, file.name);
      // 上传图片
      this.$apis.fileCommon
        .uploadFile({
          formData,
        })
        .then((res) => {
          const fileIndex = this.fileList.findIndex((f) => f.uid === file.uid);
          if (fileIndex >= 0) {
            this.fileList[fileIndex].url = res.downloadUrl;
            this.fileList[fileIndex].loading = false;
          }
          this.$emit('valChange', this.fileList);
          this.$emit('change', this.fileList);
        })
        .catch((e) => {
          console.log(e);
          const removeFile = remove(this.fileList, (f) => f.uid === file.uid);
          const msg = e.message ? `${e.message} ${removeFile[0].name}` : '系统异常';
          this.$message.error(msg);
          this.fileList = [...this.fileList];
        });
    },
    // 移除文件
    deleteImg(uid) {
      const fileIndex = this.fileList.findIndex((item) => item.uid === uid);
      if (fileIndex >= 0) {
        this.fileList.splice(fileIndex, 1);
      }
      this.$emit('valChange', this.fileList);
      this.$emit('change', this.fileList);
    },
    previewImg(index) {
      this.showIndex = index;
      this.ppVisible = true;
    },
    onImageError(index) {
      const errorFile = this.fileList.splice(index, 1);
      throw new Error(`fileList[${index}]: ${errorFile[0].url} failed to load.`);
    },
  },
};
</script>

<template>

      <!-- 主图操作区域 -->
      <div class="main-content">
        <div class="img-operate-section">
          <img
            id="dragTransparentImg"
            style="position: absolute; opacity: 0"
            :src="require('@/assets/logo.png')"
          />
          <div class="close-btn" @click.stop="_close(false)">
            <span>x</span>
          </div>
          <div class="tool-section">
            <a-icon
              type="zoom-in"
              class="common"
              title="放大"
              @click="_enlarge"
            />
            <a-icon
              type="zoom-out"
              class="common narrow"
              title="缩小"
              @click="_narrow"
            />
          </div>
          <div class="arrow-common left">
            <a-icon
              v-show="!preArrowDisabled"
              type="left-circle"
              @click.stop="_arrowClick('pre')"
            />
          </div>
          <div class="arrow-common right">
            <a-icon
              v-show="!nextArrowDisabled"
              type="right-circle"
              @click.stop="_arrowClick('next')"
            />
          </div>
          <div class="img-display">
            <img
              class="img_1"
              :draggable="true"
              :key="_img.uid"
              :id="_img.uid"
              alt=""
              :src="_img.url"
              @dragstart.stop="_imgDragStart"
              @dragover.stop="_imgDragOver"
            />
          </div>
        </div>
      </div>

      <!-- 缩略图列表 -->
      <div class="thumbnail-list">
        <ul class="ul-img-list">
          <li
            :class="i === index ? 'seled-li' : null"
            v-for="(img, i) in imgs"
            :key="i"
          >
            <img
              alt=""
              class="img_2"
              :src="img.url"
              @click.stop="_seledImg(i)"
            />
          </li>
        </ul>
      </div>

<script>
export default {
  name: "PicturePreview",
  model: {
    prop: "visible",
    event: "visibleChange",
  },
  props: {
    visible: {
      type: Boolean,
      default: false,
    },
    imgs: {
      type: Array,
      default: () => [],
    },
    showIndex: {
      type: Number,
      default: 0,
    },
  },
  data() {
    return {
      index: 0,
      preArrowDisabled: false,
      nextArrowDisabled: false,
      imgX: 0,
      imgY: 0,
      mouseX: 0,
      mouseY: 0,
      dragTransparentImg: undefined,
    };
  },
  computed: {
    _img() {
      return this.imgs.length ? this.imgs[this.index] : {};
    },
  },
  mounted() {
    this.init();
  },
  methods: {
    init() {
      this.index = this.showIndex;
      const imgsLength = this.imgs.length;
      this.nextArrowDisabled = this.index === imgsLength - 1;
      this.preArrowDisabled = this.index === 0;
      this.$nextTick(() => {
        this.dragTransparentImg = document.getElementById("dragTransparentImg");
      });
    },
    _imgDragStart(e) {
      e.dataTransfer.effectAllowed = "move";
      e.dataTransfer.setDragImage(this.dragTransparentImg, 0, 0);
      const { currentTarget, clientX, clientY } = e;
      this.imgX = currentTarget.offsetLeft;
      this.imgY = currentTarget.offsetTop;
      this.mouseX = clientX;
      this.mouseY = clientY;
    },
    _imgDragOver(e) {
      e.preventDefault();
      const { target, clientX, clientY } = e;
      target.style.top = this.imgY - (this.mouseY - clientY) + "px";
      target.style.left = this.imgX - (this.mouseX - clientX) + "px";
    },
    _enlarge() {
      const curImgDom = document.getElementById(this._img.uid);
      if (curImgDom) curImgDom.width += 80;
    },
    _narrow() {
      const curImgDom = document.getElementById(this._img.uid);
      if (curImgDom && curImgDom.width > 0) curImgDom.width -= 80;
    },
    _arrowClick(type) {
      if (type === "next") {
        if (this.nextArrowDisabled) return;
        this.index++;
        this.preArrowDisabled = false;
        this.nextArrowDisabled = this.index === this.imgs.length - 1;
      } else {
        if (this.preArrowDisabled) return;
        this.index--;
        this.nextArrowDisabled = false;
        this.preArrowDisabled = this.index === 0;
      }
    },
    _seledImg(index) {
      this.index = index;
      this.nextArrowDisabled = index === this.imgs.length - 1;
      this.preArrowDisabled = index === 0;
    },
    _close() {
      this.$emit("visibleChange", false);
    },
  },
};
</script>
