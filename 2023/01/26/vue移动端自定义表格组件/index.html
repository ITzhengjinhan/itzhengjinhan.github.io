组件一：表格组件（支持表头固定&列固定）

<template>
  <view class="table-fixed-container">
    <view class="thead">
      <view
        class="th"
        :style="{
          width: `${c.width}rpx`,
          position: c.fixed ? 'sticky' : null,
          left: c.fixed ? `${fixedWidth(c.width) * i}rpx` : null,
        }"
        v-for="(c, i) in columns"
        :key="i"
      >
        {{ c.title }}
      </view>
    </view>
    <view class="tbody">
      <view class="tr" v-for="(d, i) in data" :key="i">
        <view
          class="td"
          :style="{
            ...c.style,
            width: `${c.width}rpx`,
            position: c.fixed ? 'sticky' : null,
            left: c.fixed ? `${fixedWidth(c.width) * j}rpx` : null,
            backgroundColor: c.fixed ? '#fff' : null,
            textAlign: c.align,
          }"
          v-for="(c, j) in columns"
          :key="j"
        >
          {{ d[c.field] || '--' }}
        </view>
      </view>
    </view>
  </view>
</template>

<script>
/**
 * 表格固定（功能：表头固定&列固定）
 */
export default {
  name: 'Table',
  props: {
    columns: {
      type: Array,
      default: () => [],
    },
    data: {
      type: Array,
      default: () => [],
    },
  },
  data() {
    return {
      fixedColumns: [],
    };
  },
  mounted() {
    this.init();
    this.$nextTick(() => {
      const timer = setTimeout(() => {
        clearTimeout(timer);
        const scrollTrs = document.querySelectorAll('.scroll-cols>.tbody>.tr');
        const fixedTrs = document.querySelectorAll('.fixed-cols>.tbody>.tr');
        scrollTrs.forEach((s, i) => {
          fixedTrs[i].style.height = `${s.clientHeight}px`;
        });
      }, 500);
    });
  },
  methods: {
    init() {
      this.fixedColumns = this.columns.filter((c) => !!c.fixed);
    },
    fixedWidth(width) {
      return Number(width) + 50;
    },
  },
};
</script>

组件二：表格组件（支持单选、多选、排序、默认选中值、禁用项、编辑项）

<template>
    <view class="wyh_table">
        <view class="fixed_left" :class="{rightBorder:rightBorder}">
            <view class="th">
                <template v-if="select">
                    <view class="td tick" :class="{all:isAll}" v-if="select === 'multiple' && !Number(selectMax) && selectAll" @click="bindAll">全选</view>
                    <view class="td tick" v-else v-html="selectStr"></view>
                </template>
                <template v-for="(th, ind) in thList">
                    <view class="td" :class="{active:nSortKey==th.sortKey&&th.sortKey}" v-if="th.fixed" :key="ind" :style="{width: th.width||'220rpx'}"  @click="bindSort(th.sortKey)">
                        <view v-html="th.text"></view>
                        <view class="ic" :class="nSort" v-if="th.sortKey"></view>
                    </view>

            </view>
            <view class="tr" :class="{disabled:item.disabled, stripe: stripe && ind%2===0}" v-for="(item, ind) in list" :key="ind" @click="bindClick(item)">
                <view class="td tick" :class="{checked: isSelect(item.id)}" @click.stop="bindSelect(item, ind)" v-if="select">
                    <uni-icons type="checkbox-filled" color="#E09F70" size="18" v-if="isSelect(item.id)"></uni-icons>
                    <uni-icons type="circle" color="#999" size="18" v-else></uni-icons>

                <template v-for="(th, index) in thList">

        <view class="scroll_right">

                        <view class="td" :class="{active:nSortKey==th.sortKey&&th.sortKey}" v-if="!th.fixed" :key="ind" :style="{width: th.width||'220rpx'}" @click="bindSort(th.sortKey)">
                            <view v-html="th.text"></view>
                            <view class="ic" :class="nSort" v-if="th.sortKey"></view>

                    </template>

                <view class="tr" :class="{disabled:item.disabled, stripe: stripe && ind%2===0}" v-for="(item, ind) in list" :key="ind" @click="bindClick(item)">

<script>
    export default {
        name:"wyh-table",
        props: {
            // 表头
            thList: {
                type: Array,
                default: () => { return []}
            },
            // 是否可选
            select: {
                type: String,
                default: ''     //空  不可选    radio   单选   multiple   多选
            },
            selectMax: {
                type: Number | String,
                default: 0        //0 不限数量
            },
            // 选择列表头
            selectStr: {
                type: String,
                default: '选择'
            },
            // 是否支持全选  select 为 multiple, 并且 selectMax 有效,
            selectAll: {
                type: Boolean,
                default: false
            },
            // 列表数据
            items: {
                type: Array,
                default: () => { return []}
            },
            // 默认选中项id
            activeIds: {
                type: Array,
                default: () => { return []}
            },
            // 排序
            sort: {
                type: String,
                default: 'desc'     //desc 降序    acs 升序
            },
            //排序字段
            sortKey: {
                type: String,
                default: ''            //排序关键词
            },
            // 是否显示固定区与滚动区的分隔线
            rightBorder: {
                type: Boolean,
                default: false
            },
            // 是否展示斑马纹
            stripe: {
                type: Boolean,
                default: false
            }
        },
        data() {
            return {
                list: this.items,
                nSort: this.sort,
                nSortKey: this.sortKey,
                selectIds: this.activeIds,        //选中的ids
            };
        },
        watch: {
            items: function (ret) {
                this.list = ret;
            },
            sort: function (ret) {
                this.nSort = ret;
            },
            sortKey: function (ret) {
                this.nSortKey = ret;
            },
            activeIds: function(ret) {
                this.selectIds = ret;
            }
        },
        computed: {
            // 判断是否全选
            isAll(ret) {
                let that = this;
                let len = 0;
                let len2 = 0;
                this.list.map(function (item, index) {
                    if(!item.disabled) {
                        len += 1;
                    }
                    if(!item.disabled && that.isSelect(item.id)) {
                        len2 += 1;
                    }
                })
                if(len === len2 && len > 0) {
                    return true;
                } else {
                    return false;
                }
            }
        },
        methods: {
            bindAll() {
                let that = this;
                let arr = [];
                if(this.isAll) {
                    this.list.map(function(item,index) {
                        if( item.disabled && that.selectIds.indexOf(item.id) != -1) {
                            arr.push(item.id)
                        }
                    })
                } else {
                    this.list.map(function(item,index) {
                        if( item.disabled && that.selectIds.indexOf(item.id) != -1) {
                            arr.push(item.id)
                        }
                        if(!item.disabled) {
                            arr.push(item.id);
                        }
                    })
                }
                this.selectIds = arr;
                this.$emit('onSelect', this.selectIds);
            },
            // 排序
            bindSort(sortKey) {
                if(!sortKey) return false;
                if(this.nSortKey == sortKey) {
                    this.nSort = this.nSort == 'desc' ? 'acs' : 'desc';
                } else {
                    this.nSortKey = sortKey;
                    this.nSort = 'desc';
                }
                let sortData = {
                    sort: this.nSort,
                    sortKey: this.nSortKey
                };
                this.$emit('onSort', sortData);
            },
            // 选择
            bindSelect(item, ind) {
                if(item.disabled) {
                    return false;
                }
                if(this.select === 'radio') {
                    // 单选
                    this.selectIds = [item.id];
                } else {
                    // 多选
                    if(this.selectIds.indexOf(item.id) === -1) {
                        if(Math.floor(this.selectMax) > 0 && Math.floor(this.selectMax) === this.selectIds.length) {
                            return uni.showToast({
                                title: `最大可选数量为${Math.floor(this.selectMax)}`,
                                icon: 'none'
                            })
                        }
                        this.selectIds.push(item.id);
                    } else {
                        let _ind = this.selectIds.indexOf(item.id);
                        this.selectIds.splice(_ind, 1)
                    }
                }
                this.$emit('onSelect', this.selectIds);
            },
            // 判断是否选中
            isSelect(id) {
                if(this.selectIds.indexOf(id) === -1) {
                    return false;
                } else {
                    return true;
                }
            },
            //输入
            bindInput(ind, key){
                let obj = this.list[ind];
                let ret = {
                    key,
                    value: obj[key],
                    data: obj
                }
                this.$emit('onSetting', ret);
            },
            // 点击
            bindClick(item) {
                this.$emit('onClick', item);
            },
            // 图片预览
            previewImg(url) {
                uni.previewImage({
                    urls: [url]
                })
            }
        }
    }
</script>
