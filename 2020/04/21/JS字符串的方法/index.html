JS字符串的方法

字符串
js
发布日期:   2020-04-21

介绍：
整理目前所用过的字符串的方法，学习新增的 es6 方法。

1：charAt()返回在指定位置的字符

var str="abc"
console.log(str.charAt(0))//a

2：charCodeAt()返回在指定的位置的字符的 Unicode 编码。

var str="abc"
console.log(str.charCodeAt(1))//98，b的ASCII码

3：concat()连接字符串。

var a = "abc";
var b = "def";
var c = a.concat(b);
console.log(c);//abcdef

4：indexOf()检索字符串

var str="Hello world!"
console.log(str.indexOf("Hello"))//0
console.log(str.indexOf("World"))//-1
console.log(str.indexOf("world"))///6
//补充：search()以及includes()--其中search和indexOf用法一致，includes返回布尔值
console.log(str.search("Hello"))//0
console.log(str.includes("Hello"))//true
console.log(str.includes("Hello",1))//false,从第一个字符开始匹配

5：substring()提取字符串的片断，并在新的字符串中返回被提取的部分（左闭右开）。

var str="abc def"
console.log(str.substring(2))//c def
console.log(str.substring(2,5))// c d
//补充：substring()、substr()、slice()的区别
//substring和slice用法一致，左闭右开，而substr(n,m)表示从n开始取m位,长度为m
console.log(str.substr(2,5))//c def

6：split()把字符串分割为字符串数组。

var str="abc def ghi jkl"
console.log(str.split(" "))//["abc", "def", "ghi", "jkl"]
console.log(str.split("") )//["a", "b", "c", " ", "d", "e", "f", " ", "g", "h", "i", " ", "j", "k", "l"]
console.log(str.split(" ",3))//["abc", "def", "ghi"]
//ES6语法
var arr=[...str]//["a", "b", "c", " ", "d", "e", "f", " ", "g", "h", "i", " ", "j", "k", "l"]

7:字符串替换指定字符（串） replace：默认替换第一个符合条件的字符（串）

var str="1人1个，2人2个"
console.log(str.replace("个"，"包"))//1人1包，2人2个
console.log(str.replace(/个/g，"包"))//1人1包，2人2包
补充：match():可在字符串内检索指定的值
var str="1 abc 2 def 3"
str.match(str.match(/\d+/g/))//123

8：toLocaleLowerCase()或 toLowerCase()把字符串转换为小写。

var str="ABC def!"
console.log(str.toLocaleLowerCase())//abc def!

9：toLocaleUpperCase()或 toUpperCase()把字符串转换为大写。

var str="ABC def!"
console.log(str.toLocaleUpperCase())//ABC DEF!

10：字符串模板是增强版的字符串，用反引号（`）标识。它可以当作普通字符串使用，也可以用来定义多行字符串，或者在字符串中嵌入变量。

// 普通字符串
`In JavaScript '\n' is a line-feed.`
// 多行字符串
`In JavaScript this is
not legal.`
console.log(`string text line 1
string text line 2`);
// 字符串中嵌入变量
let name = "Bob", time = "today";
`Hello ${name}, how are you ${time}?`

11：includes(), startsWith(), endsWith();

includes()：返回布尔值，表示是否找到了参数字符串。
startsWith()：返回布尔值，表示参数字符串是否在原字符串的头部。
endsWith()：返回布尔值，表示参数字符串是否在原字符串的尾部。

let s = 'Hello world!';
s.startsWith('Hello') // true
s.endsWith('!') // true
s.includes('o') // true

12：字符串的遍历器接口 for of

for (let codePoint of 'abc') {
  console.log(codePoint)
}
// "a"
// "b"
// "c"
//与for in不同，for in遍历的键key或者下标，for of遍历是值value

13:字符串反转 […string].reverse().join(“”)

var str="的读来过倒是人明聪";
console.log([...str].reverse().join(""))//聪明人是倒过来读的

14:字符串的多次复制–repeat(n)：复制 n 次

var str="复制"
console.log(str.repeat(3))//复制复制复制

15：字符串填充到指定长度–padStart()和 padEnd();

var str1="预备开始";
console.log(str1.padStart(5,"-"))//-预备开始
var str2=131;
console.log(str2.padEnd("11","*"))//131********

16:匹配符 match()

根据正则表达式在字符串中搜索匹配项，并将匹配项作为数组返回

let str="abc ab ac"
console.log(str.match(/ab/g));//[ab,ab]

17:search()

在字符串中搜索指定值，并返回匹配的位置

var str = 'let me see';
console.log(str.search('me'));//4

18:trim()

去掉字符串前后空格
