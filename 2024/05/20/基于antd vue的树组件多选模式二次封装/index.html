基于antd vue的树组件多选模式二次封装

vue tree
vue
发布日期:   2024-05-20

背景：现有的 antd vue UI 树组件，多选模式下，有两种：一种是父子关联，即父元素选中后，对应底下所有子元素都会勾选，多个子元素取消其中一个子元素时，父元素会半勾选（此时父元素数据不会传给后端）；另一种是父子不关联，即父元素选中与底下子元素选中无关联。

需求：勾选父元素，则底下子元素也会勾选（父子关联），勾选子元素，则父元素也要勾选，而不是半选；取消子元素后，若父元素没有勾选其他子元素，则取消父元素的勾选，同时递归判断至根元素

图示

思路

首先，不能使用父子关联模式，此时 check-strictly 要设置为 true, 同时 checkable 属性为 true

此时需要动态设置 checkedData 的数据控制显示逻辑。

前提：假设后端返回一系列扁平化数据，如：

    { id: "0", name: "业务视角", desc: "描述0", order: 0 },
    { id: "1", name: "福建省烟草公司", desc: "描述1", order: 1 }
    {
        id: "0-1",
        parentId: "0",
        name: "决策与管控",
        desc: "描述0-1",
        order: 1,
    },

通过 listToTree 转为树形数据（可参考文末源码的工具方法：listToTree）

此时可通过@check=“handleCheckNode”回调中获取选中的数据，核心代码如下注释所示

handleCheckNode({ checked: checkedKeys }, { checked, node }) {
    // pathNodes：包含根级->当前节点的扁平化数据集合，更方便收集或操作相关树形数据，没有的话可用本文listToTree转换数据
    const {
    dataRef: { pathNodes, children, id: nodeId },
    } = node;

    const allChildrenNodes = [];
    let curCheckedKeys = [];
    getAllChildrenNodes(children, allChildrenNodes);
    // 获取选中节点的子孙节点(扁平)
    const getAllChildrenNodes = (nodes, allChildrenNodes = []) => {
        (nodes || []).forEach((n) => {
            const { children } = n;
            const newNode = cloneDeep(n);
            delete newNode.children;
            allChildrenNodes.push(newNode);
            if (children?.length > 0) {
                getAllChildrenNodes(children, allChildrenNodes);
            }
        });
    };
    if (checked) {
    // 选中当前节点，向下勾选当前节点底下所有子节点,同时将父级以及根基路径pathNodes拼接到数组里
    curCheckedKeys = []
        .concat(pathNodes, allChildrenNodes)
        .map((p) => p.id);
    curCheckedKeys = [].concat(checkedKeys, curCheckedKeys);
    } else {
        // 将当前节点添加为取消的节点
        allChildrenNodes.push(node);
        // 从选中的数据里过滤掉当前取消的节点
        curCheckedKeys = checkedKeys.filter((key) => {
            return !allChildrenNodes.some((a) => a.id === key);
        });
        // 递归检查并取消勾选“底下没有勾选子节点的父节点”
        curCheckedKeys = this.uncheckParentIfNoChildrenChecked(
            nodeId,
            curCheckedKeys
        );
    }
    // 可能出现重复数据，做个去重
    curCheckedKeys = uniq(curCheckedKeys);
    // 回传checkedData数据
    this.checkedData = { checked: curCheckedKeys, halfChecked: [] };
    const checkedNodes = curCheckedKeys.reduce((pre, cur) => {
        const node = this.treeList.find((ct) => ct.id === cur);
        if (node) {
            pre.push(node);
        }
        return pre;
    }, []);
    // 收集数据
    this.$emit("check", checkedNodes);
},
uncheckParentIfNoChildrenChecked(nodeId, checkedKeys) {
    const parentKey = this.getParentKey(nodeId, this.treeData);
    if (parentKey) {
    const parentChildren = this.treeList.filter(
        (item) => item.parentId === parentKey
    );
    // 判断父节点下是否有已选中的其他子节点
    const hasOtherCheckedChildren = parentChildren.some((child) =>
        checkedKeys.includes(child.id)
    );
        if (!hasOtherCheckedChildren) {
            // 若没有，则过滤掉父级
            checkedKeys = checkedKeys.filter((key) => key !== parentKey);
            // 递归检查父级的父级
            checkedKeys = this.uncheckParentIfNoChildrenChecked(
            parentKey,
            checkedKeys
            );
        }
    }
    return checkedKeys;
},

      const checkedNodes = curCheckedKeys.reduce((pre, cur) => {

        if (node) {
          pre.push(node);
        }
        return pre;

    uncheckParentIfNoChildrenChecked(nodeId, checkedKeys) {

            parentKey,
            checkedKeys
          );

      return checkedKeys;

export const listToTree = (list = [], keyName = "id", pKeyName = "pid") => {
  const newList = uniqBy(list, keyName);
  // 寻树
  const trees =
    newList.filter((l) => {
      return !newList.some((t) => t[keyName] === l[pKeyName]);
    }) || [];
  // 寻子节点
  const setChildren = (pNode) => {
    const { pathNodes } = pNode;
    const children =
      newList.filter((l) => {
        if (l[pKeyName] === pNode[keyName]) {
          l.pathNodes = pathNodes.concat({ ...l });
        }
        return l[pKeyName] === pNode[keyName];
      }) || [];
    if (children?.length) {
      pNode.children = children;
      children.forEach((node) => setChildren(node));
    } else {
      pNode.isLeaf = true;
    }
  };
  // 组装树
  trees.forEach((node) => {
    node.pathNodes = [{ ...node }];
    node.isRoot = true;
    setChildren(node);
  });
  return trees;
};
